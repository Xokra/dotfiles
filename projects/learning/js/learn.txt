function findMin(arr) {
  if (arr.length === 0) return undefined;
  let i = arr.length;
  let min = Infinity;
  while (i--) {
    if (arr[i] < min) {
      min = arr[i];
    }
  }
  return min;
}

function findMin(arr) {
    if (arr.length === 0) return undefined; // Handle empty arrays
    let min = Infinity;
    for (let i = arr.length; i--; ) {
        if (arr[i] < min) min = arr[i];
    }
    return min;
}

function findMax(arr) {
  if (arr.length === 0) return undefined; // Handle empty arrays
  let i = arr.length;
  let max = -Infinity;
  while (i--) {
    if (arr[i] > max) {
      max = arr[i];
    }
  }
  return max;
}
//with for loop
function findMax(arr) {
    if (arr.length === 0) return undefined; // Handle empty arrays
    let max = -Infinity; // Initialize with first element (saves one iteration)
    for (let i = arr.length; i--; ) {
        if (arr[i] > max) max = arr[i];
    }
    return max;
}

Map.groupBy() groups elements of an object according to string values returned from a callback function.
//creates an array
const fruits = [
  {name:"apples", quantity:300},
  {name:"bananas", quantity:500},
  {name:"oranges", quantity:200},
  {name:"kiwi", quantity:150},
]
//callback function to select low volumes
function myCallback({ quantity }) {
  return quantity > 200 ? "ok" : "low";
}
//group by ok and low
const result = Map.groupBy(fruits,myCallback);
//display results
let text ="These fruits are Ok: <br>";
for (let x of result.get("ok")) {
  text+= x.name + " " + x.quantity + "<br>";
}
text+= "<br> These fruits are low: <br>";
for (let x of result.get("low")) {
  text += x.name + " " + x.quantity + "<br>";
}
document.getElementById("demo").innerHTML = text;
//it will show
These fruits are Ok:
apples 300
bananas 500
These fruits are low:
oranges 200
kiwi 150

JS RegExp:
Regular Expression Modifiers
modifier - description
'i' perform case-insensitive mathching
let text = "Visit W3Schools";
let result = text.match(/w3schools/i)
//it will show 'W3Schools'

'g' perform a global match (find all)
let text = "Is this all there is?";
let result = text.match(/is/g);
//it will show 'is,is'

'm' perform multiline matching
let text = "\nIs th\nis it?";
let result = text.match(/^is/m);
//it will show 'is'

'd' perform start and end matching
let text = "aaaabb"; 
let result = text.match(/(aa)(bb)/d);
//it will show 'aabb,aa,bb'

Regular Expression Patterns
Brackets are used to find a range of characters:
expression - description
[abc] find any of the chacaters the backets
let text = "Is this all there is?";
let result = text.match(/[h]/g);
//it will show 'h,h'

[0-9] find any of the digits between the brackets
let text = "123456789";
let result = text.match(/[1-4]/g);
//it will show '1,2,3,4'

(x|y) find any of the alternatives separated with | 
let text = "re, green, red, green, gren, gr, blue, yellow";
let result = text.match(/(red|green)/g);
//it will show 'green.red,green'

Metacharacters are the characters with a special meaning:
Metacharacter - Description
\d find a digit
let text = "Give 100%!";
let result = text.match(/\d/g);
//it will show '1,0,0'

\s find a whitespace character
let text = "Is this all there is?";
let result = text.match(/\s/g);
//it will show ', , ,'

\b find a mtach at the beginning of a word "\bWORD", 
let text = "HELLO, LOOK AT YOU!"; 
let result = text.search(/\bLO/);
//it will show '7'
or at the end of a word "WORD\b"
let result = text.search(/LO\b/);
//it will show '3'

\uxxxx find the unicode character specified by the hexadecimal number xxxx
let text = "Visit W3Schools. Hello World!"; 
let result = text.match(/\u0057/g);
//it wil show 'W,W'

Quantifiers define quantities:
Quantifier - Description
n+ matches any string that contains at least one n
let text = "Hellooo World! Hello W3Schools!"; 
let result = text.match(/o+/g);
//it will show 'ooo,o,o,oo'

n* matches any string that contains zero or more occurrences of n
let text = "Hellooo World! Hello W3Schools!"; 
let result = text.match(/lo*/g);
//it will show 'l,looo,l,l,lo,l'

n? matches any string that contains zero or one occurrences of n
let text = "1, 100 or 1000?";
let result = text.match(/10?/g);
//it will show '1,10,10'
