function findMin(arr) {
  if (arr.length === 0) return undefined;
  let i = arr.length;
  let min = Infinity;
  while (i--) {
    if (arr[i] < min) {
      min = arr[i];
    }
  }
  return min;
}
//with for loop
function findMin(arr) {
    if (arr.length === 0) return undefined; // Handle empty arrays
    let min = Infinity;
    for (let i = arr.length; i--; ) {
        if (arr[i] < min) min = arr[i];
    }
    return min;
}

function findMax(arr) {
  if (arr.length === 0) return undefined; // Handle empty arrays
  let i = arr.length;
  let max = -Infinity;
  while (i--) {
    if (arr[i] > max) {
      max = arr[i];
    }
  }
  return max;
}
//with for loop
function findMax(arr) {
    if (arr.length === 0) return undefined; // Handle empty arrays
    let max = -Infinity; // Initialize with first element (saves one iteration)
    for (let i = arr.length; i--; ) {
        if (arr[i] > max) max = arr[i];
    }
    return max;
}

getSeconds() returns the seconds of a date as a number (0-59):
const d = new Date();
d.getSeconds();//it will show you what ever seconds you have

getMilliseconds() returns the milliseconds of a date as a number (0-999):
const d = new Date();
d.getMilliseconds();// it will show you what ever milliseconds you have

getDay() returns the weekday of adate as a number(0-6):
const days = ["Sunday", "Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday"];
const d = new Date();
let day = days[d.getDay()];// it will show what ever day you have 'thursday'

getTime() returns the number of milliseconds since January 1, 1970:
const d = new Date("1970-01-01");
d.getTime();//it will show something '0'
const d = new Date("2021-03-25");
d.getTime();// it will show '1616630400000'

Date.now() returns the number of milliseconds since January 1, 1970. but it is a static method of the Date object:
const minute = 1000 * 60;
const hour = minute * 60;
const day = hour * 24;
const year = day * 365;
let years = Math.round(Date.now() / year);

getTimezoneOffset() returns the difference (in minutes) between local time and UTC time:
let diff = d.getTimezoneOffset();
